// Secreto fuerte (32 bytes random) const payload = jwt.verify(token, SECRET, { algorithms: ['HS256'] // alg:none rechazado }); // Rol verificado en DB, no en el token const dbUser = DB.find(u => u.id === payload.userId); if (dbUser.role !== 'admin') abort();
